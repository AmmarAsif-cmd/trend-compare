#### **Trial Logic Implementation**

**Check Trial Status:**
```typescript
function isTrialActive(user: User): boolean {
  if (user.subscriptionTier !== 'trial') return false;
  if (!user.trialEndsAt) return false;
  return new Date() < user.trialEndsAt;
}

function getDaysRemaining(user: User): number {
  if (!isTrialActive(user)) return 0;
  const days = Math.ceil(
    (user.trialEndsAt!.getTime() - Date.now()) / (1000 * 60 * 60 * 24)
  );
  return days;
}
```
